program         =  statements;

statements      =  statement { statement };

statement       =  assignment
                   |  if_statement
                   |  while_loop
                   |  function_decl
                   |  function_call
                   |  return_statement;

assignment      =  "do" identifier "prirad" expression ".";

if_statement    =  "ak" "(" condition ")" "tak" block [ "inac" block ];

while_loop      =  "pokial" "(" condition ")" "tak" block;

function_decl   =  "funkcia" "(" parameter_list ")" block

return_statement =  "vrat" expression "."

parameter_list      = { identifier { "," identifier } }
argument_list       = { expression { "," expression } }

print           = "vypis" "(" expression ")" "."

block           =  "{" statement { statement } "}";
condition       =  expression { ( "==" | "!=" | ">" | "<" | ">=" | "<=" ) expression };

expression      =  term { ( "+" | "-" ) term };
term            =  factor { ( "*" | "/" | "%" | "//" ) factor };

factor          =  identifier
                   |  "-" factor
                   |  "+" factor
                   |  "(" expression ")"
                   |  literal
                   |  identifier "(" parameter_list ")";

identifier      =  ("_" | letter) {(letter | digit)};
integer         =  digit {digit};
float           =  digit {digit} "." digit {digit};
literal         =  integer | float;

letter          =  "a" | ... | "z" | "A" | ... | "Z";
digit           =  "0" | "1" | "2" | ... | "9";
